#include <stdio.h>


int main()
{
    int arr[10] = {1, 2, 3, 4, 5, 2, 3, 8, 4};     
    int unique[10];
    int i, j, count;
 
    
    for(i=0; i<10; i++)
    {
        count = 1;
        for(j=i+1; j<10; j++)
        {
            if(arr[i] == arr[j])
            {
                count++;
                unique[j] = 0;
            }
        }
 
        if(unique[i] != 0)
        {
            unique[i] = count;
        }
    }
 
   
    printf("\nUnique elements in the array are: ");
    for(i=0; i<10; i++)
    {
        if(unique[i] == 1)
        {
            printf("%d ", arr[i]);
        }
    }

    return 0;
}
